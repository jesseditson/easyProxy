var mkdirpSync = function(dir,pre){
  var pre = pre || ".",
    parts = dir.split('/'),
    top = parts.shift();
  if(top && !path.existsSync(top)){
    fs.mkdirSync(pre + "/" + top);
    if(path) mkdirpSync(parts.join('/'),top);
  }
}
